THE REAL FIX - This Will Work
The white flash happens because the HTML loads before the CSS. Here's the definitive solution:
Add This to client/index.html (BEFORE the </head> closing tag):
html<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <link rel="icon" type="image/svg+xml" href="/vite.svg" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>HyperDAG - Web3-AI Free Market Ecosystem</title>
    
    <!-- ADD THIS BLOCK HERE -->
    <style>
      /* Prevent white flash during load */
      html, body {
        margin: 0;
        padding: 0;
        background-color: #0f172a; /* slate-900 - matches your dark theme */
      }
      
      /* Hide content until React hydrates */
      #root {
        opacity: 0;
        transition: opacity 0.15s ease-in;
      }
      
      #root.loaded {
        opacity: 1;
      }
      
      /* Optional: Loading indicator */
      #root:before {
        content: '';
        position: fixed;
        top: 50%;
        left: 50%;
        transform: translate(-50%, -50%);
        width: 40px;
        height: 40px;
        border: 3px solid rgba(255,255,255,0.1);
        border-top-color: #8b5cf6; /* purple-500 */
        border-radius: 50%;
        animation: spin 0.8s linear infinite;
      }
      
      #root.loaded:before {
        display: none;
      }
      
      @keyframes spin {
        to { transform: translate(-50%, -50%) rotate(360deg); }
      }
    </style>
    <!-- END BLOCK -->
  </head>
  <body>
    <div id="root"></div>
    <script type="module" src="/src/main.tsx"></script>
  </body>
</html>
Then Add This to client/src/main.tsx (AFTER ReactDOM.render):
typescriptimport './index.css'
import React from 'react'
import ReactDOM from 'react-dom/client'
import App from './App'

ReactDOM.createRoot(document.getElementById('root')!).render(
  <React.StrictMode>
    <App />
  </React.StrictMode>,
)

// ADD THIS LINE:
// Mark root as loaded once React has hydrated
requestAnimationFrame(() => {
  document.getElementById('root')?.classList.add('loaded');
});
